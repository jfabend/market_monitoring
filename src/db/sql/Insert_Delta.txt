CREATE TABLE IF NOT EXISTS sta___tablename__ (__columnstring__);

CREATE TABLE IF NOT EXISTS tmp (__columnstring__);

COPY tmp FROM '__filepath__' WITH (FORMAT csv, HEADER true, NULL 'NA');

INSERT INTO sta___tablename__
    SELECT tmp.*
    FROM   tmp
    LEFT   JOIN sta___tablename__ USING (__idcol__)
    WHERE  sta___tablename__.__idcol__ IS NULL;

DROP TABLE IF EXISTS tmp; -- else dropped at end of session automatically
